function salvar() {

    console.log(dataNascimento.value);
    console.log(email.value);
    console.log(senha.value);
    
    const pessoa = {};
    pessoa.nome = nome.value;
    pessoa.idade = dataNascimento.value;
    pessoa.email = email.value;
    pessoa.senha = senha.value;
    console.log(pessoa);

    let pessoasCadastradas = JSON.parse(localStorage.getItem("pessoasCadastro"));
    if (!pessoasCadastradas) {
        pessoasCadastradas = [];
    }
    pessoasCadastradas.push(pessoa);
    localStorage.setItem('pessoasCadastro', JSON.stringify(pessoasCadastradas));

    return false;
}
